# 日期时间格式化

	const zeroPadding = data => (String(data).length === 1 ? `0${data}` : data);
	const dataFormatter = (date,formatter = "YYYY/MM/DD hh:mm:ss") => {
		date = new Date(date);

		let YYYY = date.getFullYear();
		let MM = zeroPadding(date.getMonth() + 1);
		let DD = zeroPadding(date.getDate());
		let hh = zeroPadding(date.getHours());
		let mm = zeroPadding(date.getMinutes());
		let ss = zeroPadding(date.getSeconds());

		const replaceDir = {
			YYYY,
			MM,
			DD,
			hh,
			mm,
			ss
		}

		return formatter.replace(/(YYYY|MM|DD|hh|mm|ss)/g,tag => replaceDir[tag] || tag);
	}
